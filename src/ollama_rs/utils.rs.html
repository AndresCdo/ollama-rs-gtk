14pub fn buffer_to_string(buffer: &TextBuffer) -> String {
15    let start_iter = buffer.start_iter();
16    let end_iter = buffer.end_iter();
17    buffer
18        .text(&start_iter, &end_iter, false)
19        .map(|s| s.to_string())
20        .unwrap_or_default()
21}

23pub fn save_file(filename: &PathBuf, text_buffer: &TextBuffer) {
24    let contents = buffer_to_string(text_buffer);
25    let mut file = File::create(filename).expect("Couldn't create file");
26    file.write_all(contents.as_bytes())
27        .expect("Couldn't write to file");
28}

30pub fn set_title(header_bar: &HeaderBar, path: &Path) {
31    if let Some(file_name) = path.file_name() {
32        let file_name: &str = &file_name.to_string_lossy();
33        header_bar.set_title(Some(file_name));
34
35        if let Some(parent) = path.parent() {
36            let subtitle: &str = &parent.to_string_lossy();
37            header_bar.set_subtitle(Some(subtitle));
38        }
39    }
40}

42pub fn open_file(filename: &Path) -> String {
43    let file = File::open(filename).expect("Couldn't open file");
44
45    let mut reader = BufReader::new(file);
46    let mut contents = String::new();
47    let _ = reader.read_to_string(&mut contents);
48
49    contents
50}

52pub fn configure_sourceview(buffer: &gtk::TextBuffer) {
53    if let Some(language_manager) = LanguageManager::default() {
54        if let Some(language) = language_manager.language("markdown") {
55            let buffer: &sourceview4::Buffer = buffer
56                .downcast_ref()
57                .expect("Failed to downcast TextBuffer to Buffer");
58            buffer.set_language(Some(&language));
59        }
60    }
61}

63#[macro_export]
64macro_rules! clone {
65    // Match `@strong` token and clone the variable
66    (@strong $($n:ident),+ => move || $body:expr) => {
67        {
68            let ($($n),+) = ($($n.clone()),+);
69            $(let $n = $n.clone();)+
70            move || $body
71        }
72    };
73    (@strong $($n:ident),+ => move |$($p:pat),*| $body:expr) => {
74        {
75            $(let $n = $n.clone();)+
76            move |$($p),*| $body
77        }
78    };
79    (@strong $($n:ident),+ => async move { $($body:tt)* }) => {
80        {
81            $(let $n = $n.clone();)+
82            async move { $($body)* }
83        }
84    };
85    // Fallback for other cases
86    ($($body:tt)*) => {
87        $($body)*
88    };
89}
